import os
import time

# LAYER 1: The Decision Layer (Your PictoBlox code)
# Since the 'Sprite' module is missing, we use a simple loop logic
print("System Ready. Waiting for commands...")

def run_sequence():
    print("Layer 1 triggered Layer 2: Starting 4-command sequence.")
    
    # LAYER 2: The Command Prompt (The Bridge)
    # We combine the 'cd' with your commands using '&&'
    # This keeps the terminal in your /2025 folder
    cmd_path = "cd /home/imed/IMEDV4/2025 && "
    
    # Triggering the 4 Layer 3 actions
    os.system(cmd_path + "./command1") 
    time.sleep(0.5)
    os.system(cmd_path + "./command2")
    time.sleep(0.5)
    os.system(cmd_path + "./command3")
    time.sleep(0.5)
    os.system(cmd_path + "./command4")
    print("Sequence Complete.")

def run_start():
    print("Layer 1 triggered Layer 2: Starting main program.")
    os.system("cd /home/imed/IMEDV4/2025 && ./start_program")

# In PictoBlox, use the 'Stage' to detect keys
while True:
    # This is the Layer 1 'Input'
    command = input("Press 'a' for sequence, 's' for start: ")
    
    if command == 'a':
        run_sequence()
    elif command == 's':
        run_start()
